링크 : [문제 바로가기](https://www.acmicpc.net/problem/10816)

```python
def solution(answers):
    answer = []
    cnt=len(answers)
    p1=[1, 2, 3, 4, 5]
    p1_answers=p1*(cnt//5) + p1[0:cnt%5]
    p2=[2, 1, 2, 3, 2, 4, 2, 5]
    p2_answers=p2*(cnt//8) + p2[0:cnt%8]
    p3=[3, 3, 1, 1, 2, 2, 4, 4, 5, 5]
    p3_answers=p3*(cnt//10) + p3[0:cnt%10]
    p1_cnt=0; p2_cnt=0; p3_cnt=0
    for i in range(cnt):
        if p1_answers[i]==answers[i]:
            p1_cnt+=1
        if p2_answers[i]==answers[i]:
            p2_cnt+=1
        if p3_answers[i]==answers[i]:
            p3_cnt+=1
    result= max(p1_cnt,p2_cnt,p3_cnt)
    if p1_cnt==result:
        answer.append(1)
    if p2_cnt==result:
        answer.append(2)
    if p3_cnt==result:
        answer.append(3)
        
    return answer
```
